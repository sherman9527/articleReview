project:
  name: Chinese AI Publishing Review System
  version: 1.0
  language: zh-CN
  type: enterprise_content_governance_platform

vision:
  description: >
    构建一个面向中文出版行业的 AI 审稿、校对、引用核验、
    敏感词治理、出版审计与多轮审核工作流平台。

  goals:
    - 自动化中文出版审稿流程
    - 降低编辑与校对成本
    - 提升出版一致性与合规性
    - 建立全链路审计能力
    - 支持多轮 AI + 人工协同审核
    - 建立长期出版知识资产

system_characteristics:
  - multi_agent
  - workflow_driven
  - audit_traceable
  - policy_based
  - version_aware
  - human_in_the_loop
  - enterprise_grade

# ============================================================
# 1. 核心业务流程
# ============================================================

publishing_workflow:

  stages:

    - id: submission
      name: 投稿阶段
      responsibilities:
        - 上传稿件
        - 文档解析
        - 元数据提取
        - 生成文档ID
        - 初始化版本

      supported_formats:
        - docx
        - pdf
        - epub
        - markdown
        - latex
        - txt

      outputs:
        - document_id
        - version_id
        - chapter_tree
        - metadata
        - extracted_references
        - extracted_images
        - extracted_tables

    - id: ai_precheck
      name: AI初审

      responsibilities:
        - 错别字检测
        - 语法检测
        - AI生成痕迹检测
        - 敏感内容检测
        - 格式检测
        - 重复内容检测

      outputs:
        - precheck_report
        - risk_score
        - sensitive_hits

    - id: first_review
      name: 一审（语言编辑）

      responsibilities:
        - 中文表达优化
        - 标点规范
        - 术语统一
        - 出版格式统一

      outputs:
        - language_review_report
        - modification_suggestions

    - id: second_review
      name: 二审（引用与事实审核）

      responsibilities:
        - 引文解析
        - DOI验证
        - ISBN验证
        - 页码核验
        - 语义一致性验证
        - 事实一致性验证

      outputs:
        - citation_report
        - hallucination_report
        - fact_check_report

    - id: third_review
      name: 三审（出版规范审核）

      responsibilities:
        - 目录检查
        - 图表编号检查
        - 参考文献格式检查
        - 页眉页脚检查
        - 图片版权检查

      outputs:
        - publication_standard_report

    - id: final_review
      name: 终审

      responsibilities:
        - 意识形态审核
        - 法律风险审核
        - 政策风险审核
        - 舆情风险审核

      outputs:
        - final_risk_report
        - publish_decision

    - id: archive
      name: 归档阶段

      responsibilities:
        - 保存最终版本
        - 保存审计日志
        - 保存diff
        - 保存电子签名
